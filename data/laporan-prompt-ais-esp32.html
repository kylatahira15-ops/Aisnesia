1. Ringkasan Proyek

Proyek ini mengintegrasikan AIS-APP (Aplikasi berbasis Express.js + MongoDB) dengan ESP32 untuk memberikan notifikasi visual (LED) dan suara (Buzzer) berdasarkan posisi kapal yang terdeteksi dalam zona geofence.

Terdapat 3 zona geofence yang masing-masing memicu LED berbeda:

ZonaWarna LEDPin ESP32Keterangan
Zona 1MerahGPIO 32Zona terluar / early warning
Zona 2KuningGPIO 25Zona menengah
Zona 3HijauGPIO 33Zona terdalam + YOLO matching

Fitur utama: Deteksi YOLO untuk mencocokkan nama kapal di Zona 3. Jika nama kapal dari AIS match (case-insensitive) dengan class name dari deteksi YOLO, maka status YOLO = 2 (match); jika YOLO aktif tapi tidak cocok = 1 (mismatch); jika YOLO tidak aktif/tidak ada deteksi = 0.

4. Pinout & Wiring ESP32

KomponenPin ESP32LogikaKeterangan
LED MERAH (Zona 1)GPIO 32Active-LOWRelay: LOW=nyala, HIGH=mati
LED KUNING (Zona 2)GPIO 25Active-LOWRelay: LOW=nyala, HIGH=mati
LED HIJAU (Zona 3)GPIO 33Active-LOWRelay: LOW=nyala, HIGH=mati
BUZZERGPIO 27Active-HIGHHIGH=bunyi, LOW=diam

Skema wiring sederhana:

ESP32 GPIO32 ──► Relay Module ──► LED MERAH (Zona 1)
ESP32 GPIO25 ──► Relay Module ──► LED KUNING (Zona 2)
ESP32 GPIO33 ──► Relay Module ──► LED HIJAU  (Zona 3)
ESP32 GPIO27 ──► Buzzer (via transistor driver jika perlu)

Catatan: Relay module active-LOW.
- LOW pada GPIO → relay aktif → LED menyala
- HIGH pada GPIO → relay non-aktif → LED mati

class Actuator {
  constructor() {
    this.zones = [];
    this.ships = [];
    this.yoloDetections = [];
    this.getShipName = null;
    this.line = '0,0,0,0';  // z1,z2,z3,yoloStatus
  }

  init(getShipNameFn) {
    this.getShipName = getShipNameFn;
    // Mulai polling YOLO tiap 2 detik
    setInterval(() => this._pollYolo(), 2000);
    // Update line tiap 1 detik
    setInterval(() => this._updateLine(), 1000);
  }

  updateZones(zones) { this.zones = zones; }
  updateShips(ships) { this.ships = ships; }

  async _pollYolo() {
    try {
      const res = await fetch('http://localhost:5000/api/detections');
      const data = await res.json();
      this.yoloDetections = data.detections || [];
    } catch (e) {
      this.yoloDetections = [];
    }
  }

  _updateLine() {
    let z1 = 0, z2 = 0, z3 = 0;
    let yoloStatus = 0;

    for (const ship of this.ships) {
      for (const zone of this.zones) {
        if (this._isInside(ship, zone)) {
          if (zone.zoneType === 'zona1') z1 = 1;
          if (zone.zoneType === 'zona2') z2 = 1;
          if (zone.zoneType === 'zona3') {
            z3 = 1;
            // YOLO matching untuk Zona 3

          }
        }
      }
    }

    this.line = `${z1},${z2},${z3},${yoloStatus}`;

9.4 arduino/arduino.ino — Firmware ESP32

#include <WiFi.h>
#include <HTTPClient.h>

const char* ssid = "your_ssid";
const char* password = "your_password";
const char* serverUrl = "http://192.168.1.100:4000/api/zones/led";

// Pin relay (active-LOW: LOW=nyala, HIGH=mati)
const int PIN_MERAH = 32;
const int PIN_KUNING = 25;
const int PIN_HIJAU = 33;
const int PIN_BUZZER = 27;

// State
int prevZ1 = -1, prevZ2 = -1, prevZ3 = -1, prevYolo = -1;
unsigned long lastPoll = 0;
unsigned long matchTimer = 0;
bool matchLedState = false;
int matchBlinkCount = 0;

void setup() {
  Serial.begin(115200);
  pinMode(PIN_MERAH, OUTPUT);
  pinMode(PIN_KUNING, OUTPUT);
  pinMode(PIN_HIJAU, OUTPUT);
  pinMode(PIN_BUZZER, OUTPUT);
  setRelay(PIN_MERAH, HIGH);
  setRelay(PIN_KUNING, HIGH);
  setRelay(PIN_HIJAU, HIGH);
  digitalWrite(PIN_BUZZER, LOW);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("WiFi connected");
}

void loop() {
  unsigned long now = millis();
  if (now - lastPoll >= 2000) {
    lastPoll = now;
    pollServer();
  }

  // State machine untuk match (yoloStatus=2)
  if (prevYolo == 2) {
    unsigned long elapsed = now - matchTimer;
    if ((elapsed / 5000) % 2 == 0) {
      // 5 detik nyala
      setRelay(PIN_HIJAU, LOW);
      digitalWrite(PIN_BUZZER, HIGH);
    } else {
      // 5 detik mati
      setRelay(PIN_HIJAU, HIGH);
      digitalWrite(PIN_BUZZER, LOW);
    }
  }
}

void pollServer() {
  HTTPClient http;
  http.begin(serverUrl);
  int code = http.GET();
  if (code == 200) {
    String body = http.getString();
    parseLine(body);
  }
  http.end();
}

void parseLine(String line) {
  // Format: "z1,z2,z3,yoloStatus"
  int z1 = line.substring(0, 1).toInt();
  int z2 = line.substring(2, 3).toInt();
  int z3 = line.substring(4, 5).toInt();
  int yolo = line.substring(6, 7).toInt();

  // Jika match (yolo==2), override z3 dan handle di state machine
  if (yolo == 2) {
    if (prevYolo != 2) {
      matchTimer = millis();
      matchLedState = false;
    }
  } else {
    // Reset match state
    matchTimer = 0;
    // Handle mismatch (yolo==1)
    if (yolo == 1 && prevYolo != 1) {
      // Buzzer 2x cepat
      for (int i = 0; i < 2; i++) {
        digitalWrite(PIN_BUZZER, HIGH);
        delay(100);
        digitalWrite(PIN_BUZZER, LOW);
        delay(100);
      }
    }
    // Set LED normal berdasarkan zone
    setRelay(PIN_MERAH, z1 ? LOW : HIGH);
    setRelay(PIN_KUNING, z2 ? LOW : HIGH);
    if (yolo != 2) setRelay(PIN_HIJAU, z3 ? LOW : HIGH);
  }

  prevZ1 = z1; prevZ2 = z2; prevZ3 = z3; prevYolo = yolo;
}

void setRelay(int pin, int state) {
  digitalWrite(pin, state);
  // state LOW = relay aktif = LED menyala
  // state HIGH = relay non-aktif = LED mati
}

Lampiran: Format Data

Format Line ESP32

z1,z2,z3,yoloStatus

Dimana:
z1 = 0/1 — Zona 1 (LED Merah) aktif
z2 = 0/1 — Zona 2 (LED Kuning) aktif
z3 = 0/1 — Zona 3 (LED Hijau) aktif
yoloStatus = 0 — YOLO tidak aktif / tidak ada deteksi
yoloStatus = 1 — YOLO aktif tapi nama TIDAK cocok (mismatch)
yoloStatus = 2 — YOLO aktif dan nama COCOK (match)

Contoh:
"1,0,0,0"   → Hanya Zona 1 aktif
"0,1,0,0"   → Hanya Zona 2 aktif
"0,0,1,1"   → Zona 3 aktif, YOLO mismatch → buzzer 2x
"0,0,1,2"   → Zona 3 aktif, YOLO match → LED blink + buzzer panjang
"1,1,1,0"   → Semua zona aktif (multi-zone), YOLO tidak aktif

Format geofences.json

{
  "_id": "...",
  "name": "Zona Pelabuhan",
  "coordinates": [[...]],   // polygon
  "zoneType": "zona1",      // "zona1" | "zona2" | "zona3" | null
  "createdAt": "...",
  "updatedAt": "..."
}
